```ts twoslash
// @noErrors
import { Bytes, Hash, Hex } from 'ox'

const message = Hex.fromString('hello world')

Hash.keccak256(message)
// @log: '0x47173285a8d7341e5e972fc677286384f802f8ef42a5ec5f03bbfa254cb01fad'

Hash.sha256(message)
// @log: '0xb94d27b9934d3e08a52e52d7da7dabfac484efe37a5380ee9088f7ace2efcde9'

Hash.ripemd160(Bytes.fromString('hello world'))
// @log: '0x98c615784ccb5fe5936fbc0cbe9dfdb408d92f0f'
```
